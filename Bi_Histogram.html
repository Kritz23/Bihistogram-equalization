def bi_hist(a):
    img=np.uint8(plt.imread( a +'.jpg'))     #Importing the Image

    #Converting into Grayscale
    #Gray = 0.2989 R + 0.5870 G + 0.1140 B

    gr = np.uint8((0.2989* img[:,:,0]) + np.uint8(0.5870 * img[:,:,1]) + np.uint8(0.1140 * img[:,:,2]))

    gr_m=gr.mean()                       # Mean of Pixel Strengths
    gr=np.array(gr)

    l_mat=[gr[i,j] if gr[i,j]<=gr_m else 0 for i in range(512) for j in range(512)]    #Creting L_Matrix if Value of Pixel is less than mean (1,262144)
    u_mat=[gr[i,j] if gr[i,j]>=gr_m else 0 for i in range(512) for j in range(512)]    #Creting U_Matrix if Value of Pixel is Greater than mean (1,262144)
    lmat=np.array(l_mat).reshape((512,512))  # Reshaving (1,262144) to (512,512)
    umat=np.array(u_mat).reshape((512,512))

    unique1, counts = np.unique(lmat, return_counts=True) #Unique1 stores all the unique elements of L_mat and counts store their counts
    lmat_d=dict(zip(unique1, counts))            #lmat_d is a dictionary of all the numbers as keys and their occureneces as their counts

    unique2, counts = np.unique(umat, return_counts=True) #Unique2 stores all the unique elements of U_mat and counts store their counts
    umat_d=dict(zip(unique2, counts))


    x0=unique1[1]           #X0  is the first non zero value in L_mat
    xm=unique1[-1]          #Xm is the last value in L_mat
    xmm=unique2[1]          #Xmm is the first non zero value in U_mat
    xl=unique2[-1]          #Xl is the last value in U_mat
    
    lmat_pr={}             #Lmat_Pr stores the probability of each unique element stored in lmat_d
    umat_pr={}             #Umat_Pr stores the probability of each unique element stored in umat_d
                 
    k=lmat_d.keys()
    k1=umat_d.keys()
    
    for i in k:                    
        lmat_pr[int(i)]=lmat_d[int(i)]/(512*512)   #Probability=No of count of that pixel/Total Number of Pixels
    
    lmat_cf=np.cumsum(list(lmat_pr.values()))      #This stores the cumulative sum of Probabilities
    
    for i in k1:
        umat_pr[int(i)]=umat_d[int(i)]/(512*512)
    
    umat_cf=np.cumsum(list(umat_pr.values()))
    
    L,U=[],[]
                 
    for i in range(len(lmat_cf)):
        L.append(np.uint8(x0+(xm-x0)*lmat_cf[i]))  #Formula as mentioned in Paper to convert Lmat Pixel values
    l_final=dict(zip(unique1,L))                   #l_final stores a dictionary in which all unique elements of Lmat and the value with which it has to be replaced is stored
    for i in range(len(umat_cf)):
        U.append(np.uint8(xmm+(xl-xmm)*umat_cf[i])) #Formula as mentioned in Paper to convert Umat Pixel values
    u_final=dict(zip(unique2,U)) 

    F=np.zeros((512,512))           #Final Numpy Matrix in which the Enhanced image will be formed
                 
    for i in range(512):            #Traversing through lmat and changing the old pixels with the new one
        for j in range(512):
            if lmat[i,j]!=0:
                F[i,j]=l_final[lmat[i,j]]
    for i in range(512):           #Traversing through lmat and changing the old pixels with the new one
        for j in range(512):
            if umat[i,j]!=0:
                F[i,j]=u_final[umat[i,j]]

                 
    fig, axes = plt.subplots(1, 2, figsize=(8, 4))   #Subplotting Original and Enhanced Image
                 
    ax = axes.ravel()
    ax[0].imshow(img)
    ax[0].set_title("Original")
    ax[1].imshow(F, cmap=plt.cm.gray)
    ax[1].set_title("Enhanced")

    plt.show()
    plt.hist(F.ravel(),256,[0,256])
    plt.show()
